    25→// Floor: large commercial tiles (60cm squares) — light cream with subtle grout
    26→function storeFloorTexture(): { map: THREE.CanvasTexture; bumpMap: THREE.CanvasTexture } {
    27→  const size = 512;
    28→  const canvas = document.createElement('canvas');
    29→  canvas.width = size; canvas.height = size;
    30→  const ctx = canvas.getContext('2d')!;
    31→  // Base cream color
    32→  ctx.fillStyle = '#e8e4d8';
    33→  ctx.fillRect(0, 0, size, size);
    34→  // Tile grid: 4 tiles per texture (each 128px = ~0.6m at scale)
    35→  const tileSize = size / 4;
    36→  for (let tx = 0; tx < 4; tx++) {
    37→    for (let ty = 0; ty < 4; ty++) {
    38→      // Subtle per-tile color variation
    39→      const v = (Math.sin(tx * 1.7 + ty * 2.3) + 1) * 0.5;
    40→      const shade = 232 + Math.floor(v * 12);
    41→      ctx.fillStyle = `rgb(${shade}, ${shade - 4}, ${shade - 16})`;
    42→      ctx.fillRect(tx * tileSize + 1, ty * tileSize + 1, tileSize - 2, tileSize - 2);
    43→      // Speckle texture (commercial tile has tiny dots)
    44→      ctx.fillStyle = 'rgba(180,170,150,0.15)';
    45→      for (let i = 0; i < 40; i++) {
    46→        ctx.fillRect(
    47→          tx * tileSize + 2 + Math.random() * (tileSize - 4),
    48→          ty * tileSize + 2 + Math.random() * (tileSize - 4),
    49→          1, 1
    50→        );
    51→      }
    52→    }
    53→  }
    54→  // Grout lines
    55→  ctx.strokeStyle = '#b8b0a0';
    56→  ctx.lineWidth = 2;
    57→  for (let i = 0; i <= 4; i++) {
    58→    ctx.beginPath(); ctx.moveTo(i * tileSize, 0); ctx.lineTo(i * tileSize, size); ctx.stroke();
    59→    ctx.beginPath(); ctx.moveTo(0, i * tileSize); ctx.lineTo(size, i * tileSize); ctx.stroke();
    60→  }
    61→  const map = new THREE.CanvasTexture(canvas);
    62→  map.wrapS = THREE.RepeatWrapping; map.wrapT = THREE.RepeatWrapping;
    63→  map.colorSpace = THREE.SRGBColorSpace; map.anisotropy = 8;
    64→
    65→  // Bump map
    66→  const bumpCanvas = document.createElement('canvas');
    67→  bumpCanvas.width = size; bumpCanvas.height = size;
    68→  const bctx = bumpCanvas.getContext('2d')!;
    69→  bctx.fillStyle = '#a0a0a0'; bctx.fillRect(0, 0, size, size);
    70→  bctx.fillStyle = '#c0c0c0';
    71→  for (let tx = 0; tx < 4; tx++) {
    72→    for (let ty = 0; ty < 4; ty++) {
    73→      bctx.fillRect(tx * tileSize + 1, ty * tileSize + 1, tileSize - 2, tileSize - 2);
    74→    }
    75→  }
    76→  bctx.fillStyle = '#505050';
    77→  for (let i = 0; i <= 4; i++) {
    78→    bctx.fillRect(i * tileSize - 1, 0, 2, size);
    79→    bctx.fillRect(0, i * tileSize - 1, size, 2);
    80→  }
    81→  const bumpMap = new THREE.CanvasTexture(bumpCanvas);
    82→  bumpMap.wrapS = THREE.RepeatWrapping; bumpMap.wrapT = THREE.RepeatWrapping;
    83→  return { map, bumpMap };
    84→}

   256→// Refrigerated glass door texture (frosty with handle)
   257→function coolerDoorTexture(): THREE.CanvasTexture {
   258→  const w = 128, h = 256;
   259→  const canvas = document.createElement('canvas');
   260→  canvas.width = w; canvas.height = h;
   261→  const ctx = canvas.getContext('2d')!;
   262→  // Frosted glass (light blue-white)
   263→  const grad = ctx.createLinearGradient(0, 0, w, h);
   264→  grad.addColorStop(0, '#c8dde8');
   265→  grad.addColorStop(0.5, '#d8e8f0');
   266→  grad.addColorStop(1, '#b8d0dc');
   267→  ctx.fillStyle = grad;
   268→  ctx.fillRect(0, 0, w, h);
   269→  // Frost pattern (random white swirls)
   270→  ctx.fillStyle = 'rgba(255,255,255,0.3)';
   271→  for (let i = 0; i < 60; i++) {
   272→    const x = Math.random() * w;
   273→    const y = Math.random() * h;
   274→    const r = 3 + Math.random() * 8;
   275→    ctx.beginPath();
   276→    ctx.arc(x, y, r, 0, Math.PI * 2);
   277→    ctx.fill();
   278→  }
   279→  // Frame (dark)
   280→  ctx.strokeStyle = '#3a3a3e';
   281→  ctx.lineWidth = 6;
   282→  ctx.strokeRect(3, 3, w - 6, h - 6);
   283→  // Handle (vertical bar on right)
   284→  ctx.fillStyle = '#5a5a62';
   285→  ctx.fillRect(w - 16, 20, 6, h - 40);
   286→  // Product silhouette behind frost (faint colored shapes)
   287→  ctx.fillStyle = 'rgba(200,100,50,0.15)';
   288→  ctx.fillRect(15, 30, 30, 50);
   289→  ctx.fillStyle = 'rgba(50,150,200,0.15)';

   295→  const tex = new THREE.CanvasTexture(canvas);
   296→  tex.colorSpace = THREE.SRGBColorSpace;
   297→  return tex;
   298→}

   300→// Fruit/vegetable texture — adds natural color variation, mottling, and
   301→// a subtle highlight to make produce look realistic instead of flat-colored.
   302→// baseColor = main color, highlightColor = lighter blush/highlight, spotColor = darker spots
   303→function fruitTexture(baseColor: string, highlightColor: string, spotColor: string, spotDensity: number = 30): THREE.CanvasTexture {
   304→  const size = 128;
   305→  const canvas = document.createElement('canvas');
   306→  canvas.width = size; canvas.height = size;
   307→  const ctx = canvas.getContext('2d')!;
   308→  // Base color
   309→  ctx.fillStyle = baseColor;
   310→  ctx.fillRect(0, 0, size, size);
   311→  // Radial highlight (glossy spot — simulates light reflection on the fruit surface)
   312→  const hgrad = ctx.createRadialGradient(size * 0.35, size * 0.3, 0, size * 0.35, size * 0.3, size * 0.5);
   313→  hgrad.addColorStop(0, highlightColor);
   314→  hgrad.addColorStop(0.5, baseColor);
   315→  hgrad.addColorStop(1, baseColor);
   316→  ctx.fillStyle = hgrad;
   317→  ctx.fillRect(0, 0, size, size);
   318→  // Natural mottling (random lighter/darker patches)
   319→  ctx.globalAlpha = 0.15;
   320→  for (let i = 0; i < 20; i++) {
   321→    ctx.fillStyle = Math.random() > 0.5 ? highlightColor : spotColor;
   322→    const x = Math.random() * size;
   323→    const y = Math.random() * size;
   324→    const r = 3 + Math.random() * 8;
   325→    ctx.beginPath();
   326→    ctx.arc(x, y, r, 0, Math.PI * 2);
   327→    ctx.fill();
   328→  }
   329→  // Small spots/specks (pores, blemishes)
   330→  ctx.globalAlpha = 0.3;
   331→  ctx.fillStyle = spotColor;
   332→  for (let i = 0; i < spotDensity; i++) {
   333→    const x = Math.random() * size;
   334→    const y = Math.random() * size;
   335→    ctx.fillRect(x, y, 1 + Math.random(), 1 + Math.random());
   336→  }
   337→  ctx.globalAlpha = 1;
   338→  const tex = new THREE.CanvasTexture(canvas);
   339→  tex.colorSpace = THREE.SRGBColorSpace;
   340→  return tex;
   341→}

   774→function buildAisle(
   775→  root: THREE.Group, num: number, category: string, z: number,
   776→  colors: string[], shelfMat: THREE.Material, shelfDarkMat: THREE.Material
   777→) {
   778→  const AISLE_LEN = 240;   // 24m long aisle
   779→  const AISLE_W = 18;      // 1.8m wide (two 0.9m shelving units back-to-back)
   780→  const AISLE_H = 22;      // 2.2m tall shelving
   781→  const SHELF_COUNT = 5;   // 5 shelves per side
   782→  const aisleX = 0;
   783→
   784→  // Aisle number sign (hanging from ceiling at the end of the aisle)
   785→  const signTex = aisleSignTexture(num, category);
   786→  const aisleSign = addShadow(new THREE.Mesh(
   787→    new THREE.PlaneGeometry(20, 7),
   788→    new THREE.MeshStandardMaterial({ map: signTex, emissive: 0xffffff, emissiveMap: signTex, emissiveIntensity: 0.3, side: THREE.DoubleSide })
   789→  ));
   790→  aisleSign.position.set(aisleX - AISLE_LEN / 2 - 5, 35, z);
   791→  aisleSign.rotation.y = Math.PI / 2;
   792→  root.add(aisleSign);
   793→
   794→  // Two shelving units (back-to-back gondola): one facing the -Z walkway, one facing the +Z walkway.
   795→  // The gondola center is at Z=z. The back panel of each unit is at the CENTER (where they meet).
   796→  // Products face OUTWARD toward the walkways on either side of the gondola.
   797→  //
   798→  // Geometry for side=-1 (lower-Z half, faces the -Z walkway):
   799→  //   shelfZ = z - AISLE_W/4 (shelf center)
   800→  //   Back panel at shelfZ + 4 (toward gondola center at z)
   801→  //   Walkway-facing front edge at shelfZ - 4 (toward the -Z walkway)
   802→  //   Products near front, labels face -Z (toward walkway)
   803→  //
   804→  // Geometry for side=+1 (upper-Z half, faces the +Z walkway):
   805→  //   shelfZ = z + AISLE_W/4 (shelf center)
   806→  //   Back panel at shelfZ - 4 (toward gondola center at z)
   807→  //   Walkway-facing front edge at shelfZ + 4 (toward the +Z walkway)
   808→  //   Products near front, labels face +Z (toward walkway)
   809→  for (const side of [-1, 1]) {
   810→    const shelfZ = z + side * (AISLE_W / 4);
   811→    // Shelving back panel — at the GONDOLA CENTER (where the two units meet)
   812→    const backPanel = addShadow(new THREE.Mesh(
   813→      new THREE.BoxGeometry(AISLE_LEN, AISLE_H, 0.5),
   814→      shelfDarkMat
   815→    ));
   816→    backPanel.position.set(aisleX, AISLE_H / 2, shelfZ - side * 4);
   817→    root.add(backPanel);
   818→
   819→    // Shelves (horizontal planks, 8 deep)
   820→    for (let s = 0; s < SHELF_COUNT; s++) {
   821→      const shelfY = 2 + s * 4.5; // shelves at y=2, 6.5, 11, 15.5, 20
   822→      const shelf = addShadow(new THREE.Mesh(
   823→        new THREE.BoxGeometry(AISLE_LEN, 0.5, 8),
   824→        shelfMat
   825→      ));
   826→      shelf.position.set(aisleX, shelfY, shelfZ);
   827→      root.add(shelf);
   828→
   829→      // Products on this shelf — organized into "facings" (contiguous blocks of
   830→      // the same product type and color, like real grocery shelves). Each facing
   831→      // is 3-5 identical products side by side. The facing layout varies per shelf
   832→      // so different shelves have different arrangements.
   833→      const productW = AISLE_LEN / 20; // base unit width (20 products worth of space)
   834→
   835→      // Facing layouts — each entry: { type, count }. The layout cycles per shelf
   836→      // so shelf 0 has one arrangement, shelf 1 another, etc.
   837→      // Types: 'box', 'bag', 'bottle', 'can', 'jar', 'pouch'
   838→      const facingLayouts = [
   839→        [{ type: 'box', count: 4 }, { type: 'can', count: 3 }, { type: 'bottle', count: 4 }, { type: 'box', count: 3 }, { type: 'jar', count: 3 }, { type: 'pouch', count: 3 }],
   840→        [{ type: 'box', count: 5 }, { type: 'bottle', count: 3 }, { type: 'can', count: 4 }, { type: 'bag', count: 3 }, { type: 'box', count: 5 }],
   841→        [{ type: 'can', count: 4 }, { type: 'box', count: 4 }, { type: 'jar', count: 3 }, { type: 'bottle', count: 3 }, { type: 'pouch', count: 3 }, { type: 'box', count: 3 }],
   842→        [{ type: 'bottle', count: 4 }, { type: 'box', count: 4 }, { type: 'bag', count: 3 }, { type: 'can', count: 3 }, { type: 'box', count: 3 }, { type: 'jar', count: 3 }],
   843→        [{ type: 'box', count: 3 }, { type: 'jar', count: 3 }, { type: 'box', count: 4 }, { type: 'bottle', count: 4 }, { type: 'pouch', count: 3 }, { type: 'can', count: 3 }],
   844→      ];
   845→      const facings = facingLayouts[s % facingLayouts.length];
   846→
   847→      // Walk along the shelf, placing each facing as a contiguous block
   848→      let posIdx = 0; // current position index along the shelf (0-19)
   849→      for (let f = 0; f < facings.length; f++) {
   850→        const facing = facings[f];
   851→        const facingColorIdx = (s * 2 + f) % colors.length;
   852→        const prodColor = colors[facingColorIdx];
   853→        const accentColor = colors[(facingColorIdx + 1) % colors.length];
   854→        const prodName = `${category.substring(0, 3).toUpperCase()}${f + 1}`;
   855→        const prodTex = productBoxTexture(prodName, prodColor, accentColor);
   856→
   857→        // Place all products in this facing (same type, same color, same texture)
